public class AngajatConcediuController : Controller
    {
        private IAngajatConcediuService _angajatConcediuService;

        public AngajatConcediuController(IAngajatConcediuService concediuService)
        {
            _angajatConcediuService = concediuService;
        }
        public async Task<IActionResult> Index()
        {
            var concediu = await _angajatConcediuService.GetAllAngajatConcediu();
            return View(concediu);
        }

        // GET: Posts/Details/5
        public async Task<IActionResult> Details(int? id)
        {
            if (id == null)
            {
                return NotFound();
            }
            var concediu = await _angajatConcediuService.GetAngajatConcediuById(id);
            if (concediu == null)
            {
                return NotFound();
            }

            return View(concediu);
        }

        // GET: Posts/Search
        public IActionResult Search(){

            return View();
        }

        // POST: Posts/Search
        [HttpPost]
        [ValidateAntiForgeryToken]
        public async Task<IActionResult> Search(string name)
        {
            var ac = await _angajatConcediuService.GetAngajatConcediuByName(name);
            return View(ac);

        }

        // GET: Posts/Create
        public IActionResult Create()
        {
            return View();
        }

        // POST: Posts/Create
        [HttpPost]
        [ValidateAntiForgeryToken]
        public async Task<IActionResult> Create([Bind("Id, TipConcediu, NrZile, AngajatiConcedii")] AngajatConcediu concediu)
        {
            if (ModelState.IsValid)
            {
                await _angajatConcediuService.CreateAngajatConcediu(concediu);
                return RedirectToAction(nameof(Index));

            }
            return View(concediu);

        }

        // GET: Posts/Edit/5
        public async Task<IActionResult> Edit(int? id)
        {
            if (id == null)
            {
                return NotFound();
            }
            var concediu = await _angajatConcediuService.GetAngajatConcediuById(id);
            if (concediu == null)
            {
                return NotFound();
            }

            return View(concediu);
        }

        // POST: Posts/Edit/5
        [HttpPost]
        [ValidateAntiForgeryToken]
        public async Task<IActionResult> Edit(int id, [Bind("Id, TipConcediu, NrZile, AngajatiConcedii")]AngajatConcediu concediu)
        {
            if (id != concediu.AngajatConcediuId)
            {
                return NotFound();
            }

            if (ModelState.IsValid)
            {
                try
                {
                    await _angajatConcediuService.UpdateAngajatConcediu(id, concediu);
                }
                catch (DbUpdateConcurrencyException)
                {
                    if (!_angajatConcediuService.AngajatConcediuExists(concediu.AngajatConcediuId))
                    {
                        return NotFound();
                    }
                    else
                    {
                        throw;
                    }
                }
                return RedirectToAction(nameof(Index));
            }

            return View(concediu);
        }

        // GET: Posts/Delete/5
        public async Task<IActionResult> Delete(int? id)
        {
            if (id == null)
            {
                return NotFound();
            }
            var concediu = await _angajatConcediuService.GetAngajatConcediuById(id);
            if (concediu == null)
            {
                return NotFound();
            }

            return View(concediu);
        }

        // POST: Posts/Delete/5
        [HttpPost]
        [ValidateAntiForgeryToken]
        public async Task<IActionResult> DeleteConfirmed(int id)
        {
            await _angajatConcediuService.DeleteAngajatConcediu(id);
            return RedirectToAction(nameof(Index));
        }
    }
